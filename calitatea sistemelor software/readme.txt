show databases  -> list of all databases
show tables from <database> -> list of tables from a specific database
create new database <name> -> new folder with the specified name in resources folder
change database to <name> -> current database is changed to <name>
create table <name>(<field1> : <type1>, <field2> : <type2> .... ) -> create a new table in the current database
insert into <name> (<val1>, <val2>, ....) -> insert new record in table <name>
delete from <name> where <cond1> and <cond2> or <cond3> .... -> delete entry fron <name> when <cond1> is true and <cond2> or <cond3> is true
update <name> set <field1> = <val1>, <field2> = <val2> ... where <cond1> and/or <cond2> ....
